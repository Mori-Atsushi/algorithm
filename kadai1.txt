#include <stdio.h>
#include <stdlib.h>

typedef struct cell {
	int item;
	struct cell *next;
} cell_t;

/*メモリ確保*/
cell_t *lalloc(void) {
	return (cell_t *)malloc(sizeof(cell_t));
}

/*先頭にデータ挿入*/
cell_t *push(int x, cell_t *top) {
	cell_t *p;
	p = lalloc();
	p->item = x;
	p->next = top;
	return(p);
}

/*先頭のデータを取り出す*/ 
cell_t *pop(cell_t *top) {
	int x;

	x = top->item;
	top = top->next;
	printf("%d, ", x);

	return (top);
}
